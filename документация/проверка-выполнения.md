# Проверка выполнения

> Запускайте только проверки, относящиеся к выбранному варианту.

```bash
# Проверка пользователей
id student
id teacher

# Проверка прав
ls -la /var/www/project

# Проверка резервных копий
ls -la /backup_*.tar.gz

# Проверка Timeshift
sudo timeshift --list

# Проверка виртуального принтера
lpinfo -v

# Проверка установленного ПО (пример)
code --version
python3 --version
docker --version  # если установлен
```
