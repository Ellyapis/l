# Вариант 1: Web-разработка (Python/JavaScript, Django/Node.js)

> Команды собраны из исходного материала без проверки на конкретной версии Ubuntu. Версии пакетов, ссылки и доступность Snap могут меняться.

## Основной способ

_Web-разработка (Python/JavaScript, Django/Node.js)_

```bash
# =====================================================
# УСТАНОВКА VS CODE
# =====================================================
sudo snap install --classic code

# =====================================================
# УСТАНОВКА NODE.JS (исправлено)
# =====================================================
curl -fsSL https://deb.nodesource.com/setup_18.x | sudo -E bash -
sudo apt install -y nodejs
node -v && npm -v

# =====================================================
# УСТАНОВКА PYTHON И DJANGO
# =====================================================
sudo apt install -y python3 python3-pip python3-venv
pip3 install django

# =====================================================
# УСТАНОВКА POSTGRESQL
# =====================================================
sudo apt install -y postgresql postgresql-contrib
sudo systemctl start postgresql
sudo systemctl enable postgresql
sudo systemctl status postgresql

# =====================================================
# УСТАНОВКА POSTMAN
# =====================================================
sudo snap install postman

# =====================================================
# УСТАНОВКА GIT
# =====================================================
sudo apt install -y git
git --version

# =====================================================
# УСТАНОВКА ДОПОЛНИТЕЛЬНЫХ ПЛАГИНОВ VS CODE
# =====================================================
code --install-extension ms-python.python
code --install-extension ms-vscode.vscode-typescript-next
```

## Способ, адаптированный для РФ

_Web-разработка (Python/JavaScript, Django/Node.js) — АДАПТИРОВАННЫЙ ДЛЯ РФ_

```bash
# =====================================================
# 1. УСТАНОВКА VS CODE (через snap - работает всегда)
# =====================================================
sudo snap install --classic code

# =====================================================
# 2. УСТАНОВКА NODE.JS через NVM (САМЫЙ НАДЕЖНЫЙ СПОСОБ)
#   (НЕ используем curl к deb.nodesource.com - заблокирован)
# =====================================================

# 2.1. Установка NVM (менеджер версий Node)
curl -o- https://raw.githubusercontent.com/nvm-sh/nvm/v0.39.7/install.sh | bash

# 2.2. Перезагрузка терминала (или выполнить source)
source ~/.bashrc

# 2.3. Проверка установки NVM
nvm --version

# 2.4. Установка Node.js версии 18 (LTS)
nvm install 18

# 2.5. Установка Node.js версии 20 (LTS) - запасной вариант
nvm install 20

# 2.6. Использование версии 18 по умолчанию
nvm use 18
nvm alias default 18

# 2.7. Проверка установки
node -v
npm -v

# =====================================================
# 3. УСТАНОВКА PYTHON И DJANGO
# =====================================================
sudo apt update
sudo apt install -y python3 python3-pip python3-venv python3-dev

# Установка Django
pip3 install django

# Проверка
python3 --version
django-admin --version

# =====================================================
# 4. УСТАНОВКА POSTGRESQL
# =====================================================
sudo apt install -y postgresql postgresql-contrib
sudo systemctl start postgresql
sudo systemctl enable postgresql
sudo systemctl status postgresql

# =====================================================
# 5. УСТАНОВКА POSTMAN (через snap - работает всегда)
# =====================================================
sudo snap install postman

# =====================================================
# 6. УСТАНОВКА GIT
# =====================================================
sudo apt install -y git
git --version

# =====================================================
# 7. УСТАНОВКА ПЛАГИНОВ VS CODE (все через внутренний магазин)
# =====================================================
code --install-extension ms-python.python
code --install-extension ms-vscode.vscode-typescript-next
code --install-extension esbenp.prettier-vscode
code --install-extension ms-vscode.vscode-js-profile-flame

# =====================================================
# 8. УСТАНОВКА ДОПОЛНИТЕЛЬНЫХ УТИЛИТ
# =====================================================

# Установка curl и wget (если не установлены)
sudo apt install -y curl wget

# Установка build-essential (для компиляции)
sudo apt install -y build-essential

# Установка Redis (опционально, для кеширования)
sudo apt install -y redis-server
sudo systemctl start redis-server

# =====================================================
# 9. ПРОВЕРКА ВСЕХ УСТАНОВОК
# =====================================================
echo "===== ПРОВЕРКА УСТАНОВКИ ====="
echo "VS Code: $(code --version | head -1)"
echo "Node.js: $(node -v)"
echo "npm: $(npm -v)"
echo "Python: $(python3 --version)"
echo "Django: $(django-admin --version 2>/dev/null || echo 'Не установлен')"
echo "PostgreSQL: $(psql --version)"
echo "Postman: $(postman --version 2>/dev/null || echo 'Установлен через snap')"
echo "Git: $(git --version)"
echo "=============================="

# =====================================================
# 10. СОЗДАНИЕ ТЕСТОВОГО ПРОЕКТА (для демонстрации)
# =====================================================

# Создаем папку проекта
mkdir -p ~/projects/demo
cd ~/projects/demo

# Создаем тестовый Django проект
django-admin startproject myproject
cd myproject

# Создаем тестовое приложение
python3 manage.py startapp myapp

# Запускаем сервер для проверки
python3 manage.py runserver 0.0.0.0:8000 &
# Открыть в браузере: http://localhost:8000

# =====================================================
# 11. АЛЬТЕРНАТИВНЫЙ СПОСОБ УСТАНОВКИ NODE.JS (через snap)
#   (если NVM не сработал - запасной вариант)
# =====================================================
# sudo snap install node --classic
```
