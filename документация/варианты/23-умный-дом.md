# Вариант 23: Умный дом (Python, C++, Zigbee, Z-Wave)

> Команды собраны из исходного материала без проверки на конкретной версии Ubuntu. Версии пакетов, ссылки и доступность Snap могут меняться.

```bash
# =====================================================
# УСТАНОВКА VS CODE
# =====================================================
sudo snap install --classic code

# =====================================================
# УСТАНОВКА PYTHON
# =====================================================
sudo apt install -y python3 python3-pip python3-venv

# =====================================================
# УСТАНОВКА HOME ASSISTANT
# =====================================================
python3 -m venv homeassistant
source homeassistant/bin/activate
pip install homeassistant
hass --open-ui

# =====================================================
# УСТАНОВКА MQTT (MOSQUITTO)
# =====================================================
sudo apt install -y mosquitto mosquitto-clients
sudo systemctl start mosquitto

# =====================================================
# УСТАНОВКА NODE.JS ДЛЯ ZIGBEE2MQTT
# =====================================================
curl -fsSL https://deb.nodesource.com/setup_18.x | sudo -E bash -
sudo apt install -y nodejs
sudo npm install -g zigbee2mqtt

# =====================================================
# УСТАНОВКА OPENZWAVE
# =====================================================
sudo apt install -y libopenzwave-dev
```
