# Вариант 14: Автоматизированное тестирование

> Команды собраны из исходного материала без проверки на конкретной версии Ubuntu. Версии пакетов, ссылки и доступность Snap могут меняться.

```bash
# =====================================================
# УСТАНОВКА VS CODE
# =====================================================
sudo snap install --classic code

# =====================================================
# УСТАНОВКА PYTHON И SELENIUM
# =====================================================
sudo apt install -y python3 python3-pip
pip3 install selenium

# =====================================================
# УСТАНОВКА CHROME DRIVER (исправлено)
# =====================================================
wget -O chromedriver.zip https://storage.googleapis.com/chrome-for-testing-public/122.0.6261.128/linux64/chromedriver-linux64.zip
sudo unzip chromedriver.zip -d /usr/local/bin/
sudo chmod +x /usr/local/bin/chromedriver

# =====================================================
# УСТАНОВКА APPIUM
# =====================================================
sudo npm install -g appium

# =====================================================
# УСТАНОВКА JMETER
# =====================================================
wget -O jmeter.zip https://downloads.apache.org/jmeter/binaries/apache-jmeter-5.6.3.zip
sudo unzip jmeter.zip -d /opt
```
