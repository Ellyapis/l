# Вариант 15: Администрирование серверов

> Команды собраны из исходного материала без проверки на конкретной версии Ubuntu. Версии пакетов, ссылки и доступность Snap могут меняться.

```bash
# =====================================================
# УСТАНОВКА ANSIBLE
# =====================================================
sudo apt install -y ansible

# =====================================================
# УСТАНОВКА ZABBIX (исправлено)
# =====================================================
wget -O zabbix-release.deb https://repo.zabbix.com/zabbix/6.0/ubuntu/pool/main/z/zabbix-release/zabbix-release_6.0-4+ubuntu22.04_all.deb
sudo dpkg -i zabbix-release.deb
sudo apt update
sudo apt install -y zabbix-server-pgsql zabbix-frontend-php php-pgsql zabbix-agent

# =====================================================
# УСТАНОВКА GRAFANA + PROMETHEUS
# =====================================================
sudo apt install -y prometheus prometheus-node-exporter grafana
sudo systemctl start grafana-server prometheus

# =====================================================
# УСТАНОВКА KVM
# =====================================================
sudo apt install -y qemu-kvm libvirt-daemon-system libvirt-clients bridge-utils virt-manager
sudo systemctl start libvirtd
```
