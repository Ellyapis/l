# Вариант 30: Транспортные системы (Python, временные ряды)

> Команды собраны из исходного материала без проверки на конкретной версии Ubuntu. Версии пакетов, ссылки и доступность Snap могут меняться.

```bash
# =====================================================
# УСТАНОВКА VS CODE
# =====================================================
sudo snap install --classic code

# =====================================================
# УСТАНОВКА PYTHON И БИБЛИОТЕК
# =====================================================
sudo apt install -y python3 python3-pip
pip3 install pandas numpy matplotlib seaborn

# =====================================================
# УСТАНОВКА PROPHET (прогнозирование)
# =====================================================
pip3 install prophet

# =====================================================
# УСТАНОВКА FOLIUM (геоданные)
# =====================================================
pip3 install folium

# =====================================================
# УСТАНОВКА POSTGRESQL + POSTGIS
# =====================================================
sudo apt install -y postgresql postgresql-15-postgis-3

# =====================================================
# УСТАНОВКА GRAFANA
# =====================================================
sudo apt install -y grafana
sudo systemctl start grafana-server

# =====================================================
# УСТАНОВКА INFLUXDB (временные ряды)
# =====================================================
sudo apt install -y influxdb
sudo systemctl start influxdb
```
