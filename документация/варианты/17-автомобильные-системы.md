# Вариант 17: Автомобильные системы (AUTOSAR, CAN-bus)

> Команды собраны из исходного материала без проверки на конкретной версии Ubuntu. Версии пакетов, ссылки и доступность Snap могут меняться.

```bash
# =====================================================
# УСТАНОВКА ECLIPSE (для AUTOSAR)
# =====================================================
sudo apt install -y eclipse

# =====================================================
# УСТАНОВКА CAN-UTILS (работа с CAN-шиной)
# =====================================================
sudo apt install -y can-utils

# =====================================================
# НАСТРОЙКА ВИРТУАЛЬНОЙ CAN-ШИНЫ
# =====================================================
sudo modprobe vcan
sudo ip link add dev vcan0 type vcan
sudo ip link set up vcan0
ip link show vcan0

# =====================================================
# УСТАНОВКА WIRESHARK (анализ трафика)
# =====================================================
sudo apt install -y wireshark
sudo dpkg-reconfigure wireshark-common
sudo usermod -aG wireshark $USER

# =====================================================
# УСТАНОВКА PYTHON-CAN
# =====================================================
pip3 install python-can

# =====================================================
# УСТАНОВКА VS CODE
# =====================================================
sudo snap install --classic code
```
