# Вариант 25: Цифровые фабрики (C#, SCADA, OPC UA)

> Команды собраны из исходного материала без проверки на конкретной версии Ubuntu. Версии пакетов, ссылки и доступность Snap могут меняться.

```bash
# =====================================================
# УСТАНОВКА VS CODE
# =====================================================
sudo snap install --classic code

# =====================================================
# УСТАНОВКА .NET SDK
# =====================================================
wget -O dotnet-install.sh https://dot.net/v1/dotnet-install.sh
chmod +x dotnet-install.sh
./dotnet-install.sh --channel 8.0
echo 'export PATH="$HOME/.dotnet:$PATH"' >> ~/.bashrc
source ~/.bashrc
dotnet --version

# =====================================================
# УСТАНОВКА POSTGRESQL
# =====================================================
sudo apt install -y postgresql

# =====================================================
# УСТАНОВКА MQTT (MOSQUITTO)
# =====================================================
sudo apt install -y mosquitto
sudo systemctl start mosquitto

# =====================================================
# УСТАНОВКА OPC UA БИБЛИОТЕКИ
# =====================================================
pip3 install opcua-asyncio

# =====================================================
# УСТАНОВКА NODE-RED (визуализация)
# =====================================================
sudo npm install -g node-red
node-red
```
