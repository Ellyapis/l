# Базовые настройки

> Общий блок для всех вариантов. Команды рассчитаны на Ubuntu/Debian и изменяют систему; перед запуском прочитайте их целиком. Пароли `123456` приведены только как учебный пример.

```bash
# =====================================================
# 1. ПРОВЕРКА ИНТЕРНЕТА И ОБНОВЛЕНИЕ СИСТЕМЫ
# =====================================================
ping -c 4 ya.ru
sudo apt update && sudo apt upgrade -y

# =====================================================
# 2. УСТАНОВКА БАЗОВОГО ПО
# =====================================================
sudo apt install -y build-essential git vim gedit curl wget net-tools
sudo apt install -y libreoffice

# =====================================================
# 3. УСТАНОВКА ВИРТУАЛЬНОГО ПРИНТЕРА (CUPS)
# =====================================================
sudo apt install -y cups cups-client cups-bsd
sudo systemctl enable cups
sudo systemctl start cups
# Проверка: открыть браузер и перейти на http://localhost:631

# =====================================================
# 4. СОЗДАНИЕ ПОЛЬЗОВАТЕЛЕЙ И ГРУПП (ИСПРАВЛЕНО)
# =====================================================
sudo groupadd devgroup
sudo useradd -m -G devgroup -s /bin/bash student
sudo useradd -m -G devgroup -s /bin/bash teacher
echo "student:123456" | sudo chpasswd
echo "teacher:123456" | sudo chpasswd

# Проверка создания
id student
id teacher

# =====================================================
# 5. НАСТРОЙКА ПРАВ ДОСТУПА
# =====================================================
sudo mkdir -p /var/www/project
sudo chown root:devgroup /var/www/project
sudo chmod 770 /var/www/project

# Проверка прав
ls -la /var/www/

# =====================================================
# 6. РЕЗЕРВНОЕ КОПИРОВАНИЕ
# =====================================================
sudo tar -czf /backup_home_$(date +%Y%m%d).tar.gz /home/student 2>/dev/null
sudo tar -czf /backup_etc_$(date +%Y%m%d).tar.gz /etc 2>/dev/null
ls -la /backup_*.tar.gz

# =====================================================
# 7. УСТАНОВКА TIMESHIFT (ТОЧКИ ВОССТАНОВЛЕНИЯ)
# =====================================================
sudo apt install -y timeshift
sudo timeshift --create --comments "Initial_Backup_$(date +%Y%m%d)"
sudo timeshift --list

# =====================================================
# 8. НАСТРОЙКА ЖУРНАЛОВ МОНИТОРИНГА
# =====================================================
sudo journalctl -f
# Ctrl+C для выхода

# =====================================================
# 9. ДОПОЛНИТЕЛЬНО: НАСТРОЙКА SSH (для удаленного доступа)
# =====================================================
sudo apt install -y openssh-server
sudo systemctl enable ssh
sudo systemctl start ssh
sudo systemctl status ssh
```
