# Вариант 19: Многопользовательские онлайн-игры (C++, Python)

> Команды собраны из исходного материала без проверки на конкретной версии Ubuntu. Версии пакетов, ссылки и доступность Snap могут меняться.

```bash
# =====================================================
# УСТАНОВКА VS CODE
# =====================================================
sudo snap install --classic code

# =====================================================
# УСТАНОВКА C++ И PYTHON
# =====================================================
sudo apt install -y build-essential python3 python3-pip

# =====================================================
# УСТАНОВКА PROMETHEUS (мониторинг)
# =====================================================
sudo apt install -y prometheus prometheus-node-exporter
sudo systemctl start prometheus

# =====================================================
# УСТАНОВКА GRAFANA
# =====================================================
sudo apt install -y software-properties-common
sudo add-apt-repository "deb https://packages.grafana.com/oss/deb stable main"
wget -q -O - https://packages.grafana.com/gpg.key | sudo apt-key add -
sudo apt update
sudo apt install -y grafana
sudo systemctl start grafana-server
sudo systemctl enable grafana-server

# =====================================================
# УСТАНОВКА HAProxy (балансировка)
# =====================================================
sudo apt install -y haproxy
sudo systemctl start haproxy

# =====================================================
# УСТАНОВКА LOCUST (нагрузочное тестирование)
# =====================================================
pip3 install locust
```
