# Вариант 20: Компьютерное зрение (Python, C++, CUDA)

> Команды собраны из исходного материала без проверки на конкретной версии Ubuntu. Версии пакетов, ссылки и доступность Snap могут меняться.

```bash
# =====================================================
# УСТАНОВКА VS CODE
# =====================================================
sudo snap install --classic code

# =====================================================
# УСТАНОВКА PYTHON И OPENCV
# =====================================================
sudo apt install -y python3 python3-pip
pip3 install opencv-python opencv-contrib-python

# =====================================================
# УСТАНОВКА CUDA (для NVIDIA GPU)
# =====================================================
sudo apt install -y nvidia-cuda-toolkit nvidia-driver-535
nvidia-smi

# =====================================================
# УСТАНОВКА V4L2 (работа с камерами)
# =====================================================
sudo apt install -y v4l-utils

# =====================================================
# УСТАНОВКА YOLO (детекция объектов)
# =====================================================
git clone https://github.com/pjreddie/darknet
cd darknet
make
./darknet detect cfg/yolov3.cfg yolov3.weights data/dog.jpg

# =====================================================
# УСТАНОВКА TENSORFLOW
# =====================================================
pip3 install tensorflow
```
